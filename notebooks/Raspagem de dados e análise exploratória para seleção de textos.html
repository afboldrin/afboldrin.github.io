import csv
from google.colab import files

def obter_palavras_do_arquivo():
    # Realize o upload do arquivo
    arquivo_upload = files.upload()

    # Obtenha o nome e o tipo do arquivo
    nome_arquivo = list(arquivo_upload.keys())[0]
    tipo_arquivo = nome_arquivo.split('.')[-1]

    try:
        # Verifique se o tipo de arquivo  suportado
        if tipo_arquivo == 'txt':
            with open(nome_arquivo, 'r') as arquivo:
                # Leia as palavras do arquivo de texto
                palavras = arquivo.read().split()
        else:
            print('Tipo de arquivo no suportado. Por favor, faa o upload de um arquivo CSV ou TXT.')
            return None

    except Exception as e:
        print(f'Erro ao ler o arquivo: {e}')
        return None

    return palavras

def get_ebook_link(book_number, n):
    """
    Retorna uma lista de URLs para download de ebooks do Projeto Gutenberg com base no nmero do livro.

    Parmetros:
    - book_number: Nmero do livro no Projeto Gutenberg.
    - n: ndice para escolher entre as URLs disponveis (0, 1, ou 2).

    Retorna:
    Lista de URLs correspondentes.
    """
    subdirectories = '0' if 1 <= book_number <= 9 else '/'.join(str(book_number)[:-1])
    prefix = "https://mirrors.xmission.com/gutenberg"

    def generate_link(suffix):
        return f'{prefix}/{subdirectories}/{book_number}/{book_number}{suffix}.txt'

    lst = [generate_link(""), generate_link("-0"), generate_link("-8")]
    return lst[n]

class MyCorpusReader(PlaintextCorpusReader):
    def contar_palavras(self, fileid):
        """
        Conta o nmero de palavras em um arquivo no corpus.
        """
        palavras = self.words(fileid)
        return len(palavras)

    def TTR(self, fileid):
        """
        Calcula a Taxa Tipo-Token (TTR) para um arquivo no corpus.
        """
        palavras_no_arquivo = self.words(fileid)
        num_tipos = len(set(palavras_no_arquivo))
        num_tokens = len(palavras_no_arquivo)
        ttr = num_tipos / num_tokens
        return ttr

    def calcular_palavras_unicas(self, fileid, granularidade='palavras'):
        """
        Calcula o nmero de palavras nicas em um arquivo do corpus.
        """
        palavras_no_arquivo = self.words(fileid)
        palavras_unicas = set(palavras_no_arquivo)
        return len(palavras_unicas)

        Calcula o nmero de palavras nicas em um arquivo do corpus que no esto no vocabulrio prvio.

def calculate_ttr(data, granularity='words'):
    """
    Calculate Type-Token Ratio (TTR) based on the chosen granularity.

    Parameters:
    - data: List of words, sentences, or paragraphs.
    - granularity: Granularity for TTR calculation ('words', 'sentences', 'paras').
    """

    if granularity == 'words':
        pass  # No additional processing needed for words
    elif granularity == 'sentences':
        # Flatten the list of sentences to get a list of words
        data = [word for sentence in data for word in sentence]
    elif granularity == 'paras':
        # Flatten the list of paragraphs to get a list of sentences
        data = [sentence for paragraph in data for sentence in paragraph]
        # Flatten the list of sentences to get a list of words
        data = [word for sentence in data for word in sentence]
    else:
        raise ValueError("Invalid granularity. Choose 'words', 'sentences', or 'paras'.")

    # Calculate the number of unique words (types) and the total number of words (tokens)
    num_types = len(set(data))
    num_tokens = len(data)

    # Calculate TTR
    ttr = num_types / num_tokens

    return ttr

def greedy(fileids, vocab_previo):
    vocab_novo_count = {}
    for fileid in fileids:
        vocab_novo_count[fileid] = 0
    lowered_vocab = set(word.lower() for word in vocab_previo)
    for fileid in fileids:
        palavras_unicas = set(word.lower() for word in my_corpus.words(fileid))
        total_words = len(my_corpus.words(fileid))

        for word in palavras_unicas:
            if word not in lowered_vocab:
                vocab_novo_count[fileid] += 1
    ordered_books = dict(sorted(vocab_novo_count.items(), key=lambda item: item[1], reverse=True))
    return ordered_books
