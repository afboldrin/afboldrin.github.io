from sklearn.linear_model import LinearRegression
import numpy as np
import matplotlib.pyplot as plt

# Independent variable
X = df_m[['Interest']].values  # Independent variable (reshaped to be 2D)
y = df_m['Price'].values       # Dependent variable

# Create the linear regression model
model = LinearRegression()

# Fit the model
model.fit(X, y)

# Get the coefficients
intercept = model.intercept_
slope = model.coef_[0]

print(f'Intercept: {intercept}')
print(f'Slope: {slope}')

# Make predictions
y_pred = model.predict(X)

# Optional: Plot the results
plt.scatter(X, y, color='blue',alpha=0.8)
plt.plot(X, y_pred, color='black', linewidth=2)
plt.xlabel('Interest')
plt.ylabel('Price')
plt.title('Linear Regression: Interest vs. Price')
plt.show()
